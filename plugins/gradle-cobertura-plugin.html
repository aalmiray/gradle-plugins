Summary

Apply code coverage to your project using Cobertura.

Website https://github.com/stevesaliman/gradle-cobertura-plugin
Author stevesaliman
Version 2.2.2

Usage

buildscript {
    repositories {
        jcenter()
    }

    dependencies {
        classpath 'net.saliman:gradle-cobertura-plugin:2.2.2'
    }
}

apply plugin: 'cobertura'
