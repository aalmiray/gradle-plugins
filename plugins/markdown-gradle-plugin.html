Summary

Provides a facility for converting Markdown into HTML, as well as converting HTML back into Markdown.

Website https://github.com/aalmiray/markdown-gradle-plugin
Author aalmiray
Version 0.1

Usage

buildscript {
    repositories {
        jcenter()
        maven { url 'http://dl.bintray.com/content/aalmiray/kordamp' }
    }

    dependencies {
        classpath 'org.kordamp:markdown-gradle-plugin:0.1'
    }
}

apply plugin: 'markdown'
