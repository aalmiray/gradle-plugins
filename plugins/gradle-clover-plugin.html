Summary

The plugin provides generation of code coverage reports using Clover.

Website https://github.com/bmuschko/gradle-clover-plugin
Author bmuschko
Version 0.8.1

Usage

buildscript {
    repositories { jcenter() }
    dependencies {
        classpath 'org.gradle.api.plugins:gradle-nexus-plugin:0.8.1'
    }
}

apply plugin: 'clover'

dependencies {
    clover 'com.cenqua.clover:clover:3.2.0'
}

clover {
    classesBackupDir = file("${sourceSets.main.classesDir}-backup")
    licenseLocation = 'clover-license.txt'
}
