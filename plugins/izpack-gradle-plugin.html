Summary

The plugin provides a one-stop solution for packaging, distributing and deploying applications for the Java platform using IzPack.

Website https://github.com/bmuschko/gradle-izpack-plugin
Author bmuschko
Version 0.2.3

Usage

buildscript {
    repositories {
        jcenter()
    }

    dependencies {
        classpath 'org.gradle.api.plugins:gradle-izpack-plugin:0.2.3'
    }
}

apply plugin: 'izpack'

dependencies {
    izpack 'org.codehaus.izpack:izpack-standalone-compiler:4.3.4'
}

izpack {
    baseDir = file("$buildDir/assemble/izpack")
    installFile = file('installer/izpack/installer.xml')
    outputFile = file("$buildDir/distributions/${project.name}-${version}-installer.jar")
    compression = 'deflate'
    compressionLevel = 9
}
